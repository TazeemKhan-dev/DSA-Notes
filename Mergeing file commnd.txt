# ===========================================

# ===========================================

$sourceRoot = "."

# Create destination folder if not exists
if (!(Test-Path ".\Merged-Files")) {
    New-Item -ItemType Directory -Path ".\Merged-Files" | Out-Null
}

# Create / reset master file
"" | Out-File $destinationFile

$folders = Get-ChildItem -Directory

foreach ($folder in $folders) {
    Write-Host "Adding section for $($folder.Name)..."

    # Write folder header
    "## üìÅ $($folder.Name)`n" | Out-File $destinationFile -Append

    # Append all markdown files inside folder
    Get-Content "$($folder.FullName)\*.md" | Out-File $destinationFile -Append

    # Blank line after each folder
    "`n`n" | Out-File $destinationFile -Append
}

Write-Host "`n‚û§ MASTER DSA FILE CREATED: Merged-Files\MASTER-DSA-NOTES.md"
